
4. **`src/protocol/frame.rs` â€” `recv_frame` integration**:
   ```

   if header.has_unknown_flags() { return Err(UnknownFlags) }
   wire_payload = read(header.length);

       decompress_payload(&wire_payload)?
   } else {
       wire_payload
   };

   bincode::serde::decode_from_slice::<Message, _>(&payload, config)
   ```
