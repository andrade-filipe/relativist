## Deliverable

`relativist-core::encoding::registry` module exporting:

```rust
pub struct EncoderRegistry {
    codecs: HashMap<String, Box<dyn Codec>>,
}

#[derive(Debug, thiserror::Error, PartialEq, Eq)]
pub enum RegistryError {
    #[error("encoder '{0}' is already registered")]
    DuplicateName(String),
    #[error("encoder '{0}' not found")]
    NotFound(String),
    #[error("encode failed: {0}")]
    Encode(#[from] EncodeError),
    #[error("decode failed: {0}")]
    Decode(#[from] DecodeError),
}

impl EncoderRegistry {
    pub fn new() -> Self;
    pub fn register(&mut self, codec: Box<dyn Codec>) -> Result<(), RegistryError>;
    pub fn get(&self, name: &str) -> Option<&dyn Codec>;
    pub fn list(&self) -> Vec<(&str, &str)>;  // (name, description)
    pub fn encode_and_validate(&self, name: &str, input: &[u8]) -> Result<Net, RegistryError>;
    pub fn decode(&self, name: &str, net: &Net) -> Result<serde_json::Value, RegistryError>;
}

impl Default for EncoderRegistry { /* delegates to new() */ }
```

`encode_and_validate` MUST call `codec.encode(input)?` then `validate_encoded_net(&net)?`,
returning the same `Net` on success.

`list()` returns sorted by name for deterministic CLI output.
