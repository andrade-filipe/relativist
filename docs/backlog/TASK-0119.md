## Context

This task validates the entire CLI pipeline end-to-end for local mode: generate a network, run local mode reduction, verify the output file can be deserialized, and verify metrics are written. It exercises the full chain: CLI parsing -> config mapping -> net I/O -> grid loop -> output. This is the smoke test that proves Phase 6 wiring is correct.

The test uses `assert_cmd` patterns (invoke the binary as a subprocess) or calls the command functions directly in an integration test.

```rust
// In tests/cli_integration.rs

use relativist::config::{LocalArgs, GenerateArgs};
use relativist::io::{load_net_from_file, deserialize_net};
use std::path::PathBuf;
use tempfile::TempDir;

#[test]
fn test_generate_and_local_round_trip() {
    let tmp = TempDir::new().unwrap();
    let net_path = tmp.path().join("test.bin");
    let output_path = tmp.path().join("result.bin");
    let metrics_path = tmp.path().join("metrics.json");

    // Generate a small network
    let gen_args = GenerateArgs {
        workload: "era-chain".to_string(),  // simplest workload
        size: 10,
        output: net_path.clone(),
    };
    run_generate_command(gen_args).unwrap();
    assert!(net_path.exists());

    // Run local mode
    let local_args = LocalArgs {
        workers: 2,
        net: net_path,
        max_rounds: Some(100),
        output: Some(output_path.clone()),
        metrics: Some(metrics_path.clone()),
        strategy: "round-robin".to_string(),
    };
    run_local_command(local_args).unwrap();

    // Verify output
    let result_net = load_net_from_file(&output_path).unwrap();
    assert!(result_net.redex_queue.is_empty(), "should be in Normal Form");

    // Verify metrics
    let metrics_json = std::fs::read_to_string(&metrics_path).unwrap();
    let _: serde_json::Value = serde_json::from_str(&metrics_json).unwrap();
}
```
